# 九九乘法实现方法

for i in range(1,10):
    line = ' '
    for j in range(1,10):
        if j <= i:
            line += str(j) + '*' + str(i) + '=' + str(i*j) + ' '
    print(line)


for i in range(1,10):
    for j in range(1,i+1):
        print('{}*{}={}'.format(j,i,i*j,2 if j<2 else 3),end=' ')

# 菱形实现方法

for i in range(4,-5,-1):
    print(abs(i)*' ',end=(''))
    for j in range(-4,5):
        if abs(i)+abs(j)<5:
            print('*',end=(''))
    print()

#求100以内的质数

#三个数做比较排序(从大到小排序)

out = None
for i in range(3):
    nums.append(int(input('{} = '.format(i))))

if nums[0] > nums[1]:
    if nums[0] > nums[2]:
        if nums[1] > nums[2]:
            out = [0,1,2]
        else:
            out = [0,2,1]
    else:
        out = [2,0,1]
else:
    if nums[0] < nums[2]:
        if nums[1] < nums[2]:
            out = [2,1,0]
        else:
            out = [1,2,0]
    else:
        out = [1,0,2]

for i in out:
    print(nums[i])
